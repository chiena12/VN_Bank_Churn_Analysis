Bước 2: Tối ưu tham số bằng Optuna (Bayesian Optimization)
Tư duy Senior: Thay vì "ném lao may rủi" như RandomSearch, ta dùng Optuna để xây dựng một mô hình xác suất về các tham số. Nó học từ những lần thử thất bại để tập trung vào vùng tham số tiềm năng.

Python
import optuna
from lightgbm import LGBMClassifier
from sklearn.model_selection import cross_val_score

def objective(trial):
    # 1. Định nghĩa không gian tìm kiếm (Search Space)
    param = {
        'n_estimators': trial.suggest_int('n_estimators', 100, 1000),
        'learning_rate': trial.suggest_float('learning_rate', 0.01, 0.1, log=True),
        'num_leaves': trial.suggest_int('num_leaves', 20, 150),
        'max_depth': trial.suggest_int('max_depth', 4, 12),
        'min_child_samples': trial.suggest_int('min_child_samples', 20, 100),
        'scale_pos_weight': trial.suggest_float('scale_pos_weight', 1, 6), # Xử lý imbalance
        'lambda_l1': trial.suggest_float('lambda_l1', 1e-8, 10.0, log=True),
        'verbosity': -1
    }
    
    # 2. Khởi tạo mô hình
    model = LGBMClassifier(**param, random_state=42)
    
    # 3. Sử dụng Cross-validation để tránh Overfitting
    # Senior Tip: Tối ưu dựa trên f1_macro hoặc f1 của lớp 1
    score = cross_val_score(model, x_train, y_train, cv=5, scoring='f1').mean()
    return score

# 4. Chạy tối ưu
study = optuna.create_study(direction='maximize')
study.optimize(objective, n_trials=100) # Tăng n_trials nếu máy mạnh

print(f"Best F1 Score: {study.best_value}")
print(f"Best Params: {study.best_params}")
Bước 3: Tối ưu ngưỡng dự báo (Threshold Tuning)
Tư duy Senior: Mặc định mô hình cắt ở 0.5. Nhưng trong ngân hàng, chi phí bỏ sót khách rời đi (Recall thấp) khác hoàn toàn chi phí tặng quà nhầm (Precision thấp). Ta cần tìm một "điểm ngọt" về mặt kinh tế.

Python
from sklearn.metrics import precision_recall_curve, f1_score
import numpy as np

# 1. Lấy xác suất thay vì nhãn cứng
y_probs = model.predict_proba(x_test)[:, 1]

# 2. Tính toán Precision và Recall tại các ngưỡng khác nhau
precisions, recalls, thresholds = precision_recall_curve(y_test, y_probs)

# 3. Tìm ngưỡng tối ưu hóa F1-score
f1_scores = 2 * (precisions * recalls) / (precisions + recalls)
best_threshold = thresholds[np.argmax(f1_scores)]

print(f"Ngưỡng tối ưu để đạt F1 cao nhất: {best_threshold:.4f}")

# 4. Áp dụng ngưỡng mới
y_pred_new = (y_probs >= best_threshold).astype(int)

Bước 4: Ensemble Stacking (Kết hợp đa mô hình)
Tư duy Senior: Không có mô hình nào là hoàn hảo. LightGBM mạnh về tốc độ, XGBoost mạnh về độ chính xác sâu, CatBoost cực tốt với dữ liệu phân loại. Kết hợp chúng lại giúp giảm sai số hệ thống của từng loại.

Python
from sklearn.ensemble import StackingClassifier
from xgboost import XGBClassifier
from catboost import CatBoostClassifier
from sklearn.linear_model import LogisticRegression

# 1. Định nghĩa các mô hình cơ sở (Base Models)
# Senior Tip: Sử dụng tham số tốt nhất tìm được từ Optuna ở bước 2
base_models = [
    ('lgbm', LGBMClassifier(**study.best_params, random_state=42)),
    ('xgb', XGBClassifier(n_estimators=500, learning_rate=0.05, max_depth=6, random_state=42)),
    ('cat', CatBoostClassifier(iterations=500, depth=6, verbose=0))
]

# 2. Định nghĩa mô hình tổng hợp (Meta-learner)
# Ta dùng LogisticRegression để học cách 'trọng dụng' kết quả của 3 ông trên
stack_model = StackingClassifier(
    estimators=base_models,
    final_estimator=LogisticRegression(),
    cv=5
)

# 3. Huấn luyện
stack_model.fit(x_train, y_train)

# 4. Đánh giá
y_pred_stack = stack_model.predict(x_test)
print(classification_report(y_test, y_pred_stack))
